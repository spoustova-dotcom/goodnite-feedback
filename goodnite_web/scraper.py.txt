import pandas as pd
import time
from playwright.sync_api import sync_playwright

def scrape_booking():
    # Načtení tvých odkazů
    df_links = pd.read_csv('apartments_links_clean.csv')
    results = []

    with sync_playwright() as p:
        # Spuštění prohlížeče
        browser = p.chromium.launch(headless=True)
        page = browser.new_page()

        for index, row in df_links.iterrows():
            name = row['Apartmán']
            url = row['Booking URL']
            
            if pd.isna(url) or "booking.com" not in str(url):
                continue
                
            print(f"Scrapuji: {name}...")
            try:
                page.goto(url, wait_until="networkidle")
                # Získání celkového hodnocení
                rating = page.locator('[data-testid="review-score-component"] div:first-child').first.inner_text()
                
                # Zde by pokračovala logika pro rozkliknutí recenzí a sběr textů
                results.append({
                    "Datum": time.strftime("%d.%m.%Y"),
                    "apartmán": name,
                    "hodnoceni": rating.replace(',', '.'),
                    "Platforma": "Booking"
                })
            except Exception as e:
                print(f"Chyba u {name}: {e}")

        browser.close()
    
    # Uložení výsledků do CSV (přidání k existujícím)
    new_data = pd.DataFrame(results)
    new_data.to_csv('Recenze ⭐ - Text_recenze.csv', mode='a', header=False, index=False)
    print("Hotovo! Data byla připsána do souboru.")

if __name__ == "__main__":
    scrape_booking()
